def reverse_rol1(byte, shift):
    # 循环左移的逆操作是循环右移
    return (byte >> shift) | ((byte << (8 - shift)) & 0xff)

def decrypt_byte(encrypted_byte, idx, v26, v27):
    # 计算位移量 v29
    # 由于 v29 的计算涉及到魔数乘法，这里假设我们可以直接计算
    # 具体实现需要根据加密函数的上下文
    # 以下是一个简化的计算方式，具体需要根据实际情况调整
    v29 = (0xCCCCCCCCCCCCCCCD * idx) >> 64  # 假设 idx 是 i 的某种表示
    v29 = (v29 & 0xff) >> 2

    # 假设 v30 = __ROL1__(byte ^ (idx + 90), idx - 5 * v29)
    # 我们需要逆向计算 shift = idx - 5 * v29
    shift = (idx - 5 * v29) % 8  # 确保 shift 在 0-7 之间
    decrypted_byte = reverse_rol1(encrypted_byte, shift)

    # 假设 v30 = (byte ^ (idx + 90))
    # 所以原始字节为 decrypted_byte ^ (idx + 90)
    original_byte = decrypted_byte ^ (idx + 90)

def xxtea_decrypt(v, n, key):
    """Decrypt data using XXTEA algorithm."""
    delta = 0x9e3779b9
    r = 6 + 52 // n
    total = delta * r

    for _ in range(r):
        e = (total >> 2) & 3
        for j in range(n - 1, 0, -1):
            v1 = v[j - 1]
            part1 = (v1 >> 5) ^ (v[j] << 2)
            part2 = (v[j] >> 3) ^ (v1 << 4)
            part3 = (total ^ v[j]) + (key[(j & 3) ^ e] ^ v1)

    key = "ISCC{s_ale_ru_upatu_prrlaullre_}"
    decrypted_chars = []

        k = ord(key[i])

def _hex_from_cipher(cipher: str) -> str:
    """字符转为4位十六进制，去掉前导 '00'（A.c 的逆操作）"""
    hex_string = ''.join(f'{ord(char):04x}' for char in cipher)
    return hex_string[2:] if hex_string.startswith('00') else hex_string


def _undo_padding(hex_str: str) -> str:
    """
    去掉每4位中插入的 '00'，还原插针前的串（A.b 的逆操作）
    - 如果长度 % 4 == 2，说明最后一个 '00' 被截断了，需补回。
    """
    if len(hex_str) % 4 == 2:
        hex_str += '00'
    return ''.join(hex_str[i:i + 2] for i in range(0, len(hex_str), 4))


def _undo_pair_shuffle(replaced: str) -> str:
    """
    还原2字节或4字节片段的原始顺序。
    - 4字节片段格式为：c2 c1 2 1
    - 当片段末尾是 '21'，说明前两个字符是反转的。
    """
    result = []
    i = 0
    while i < len(replaced):
        if i + 3 < len(replaced) and replaced[i + 2:i + 4] == '21':
            c2, c1 = replaced[i], replaced[i + 1]
            result.extend([c1, c2])
            i += 4
        else:
            result.extend(replaced[i:i + 2])
            i += 2
    return ''.join(result)


def _restore_zero_positions(segment: str, base_idx: int) -> str:
    """
    恢复被替换为 '3' 的 '0'。
    - base_idx = 0：处理 sb 段（偶数位）
    - base_idx = 1：处理 sb2 段（奇数位）
    """
    return ''.join(
        '0' if char == '3' and (idx == base_idx or (idx - base_idx) % 3 == 0) else char
        for idx, char in enumerate(segment)
    )


def _split_and_fix_zero(concat: str) -> str:
    """
    分割成 sb2 和 sb，修复被替换为 '3' 的零位，并交错合并。
    """
    half = len(concat) // 2
    sb2 = _restore_zero_positions(concat[:half], base_idx=1)
    sb = _restore_zero_positions(concat[half:], base_idx=0)

    result = []
    even_idx = odd_idx = 0
    for i in range(len(concat)):
        if i % 2 == 0:
            result.append(sb[even_idx])
            even_idx += 1
        else:
            result.append(sb2[odd_idx])
            odd_idx += 1
    return ''.join(result)


def _hex_to_plain(hex_str: str) -> str:
    """
    十六进制还原为字符：
    - 以 '0' 开头代表 3 位组（0xx）➜ 单字节
    - 否则为 4 位组（xxxx）➜ Unicode 字符
    """
    plain_chars = []
    i = 0
    while i < len(hex_str):
        if hex_str[i] == '0' and i + 2 < len(hex_str):
            plain_chars.append(chr(int(hex_str[i + 1:i + 3], 16)))
            i += 3
        else:
            plain_chars.append(chr(int(hex_str[i:i + 4], 16)))
            i += 4
    return ''.join(plain_chars)


def decrypt(cipher: str) -> str:
    """主解密流程入口"""
    hex_stage = _hex_from_cipher(cipher)
    unpadded = _undo_padding(hex_stage)
    unshuffled = _undo_pair_shuffle(unpadded)
    restored = _split_and_fix_zero(unshuffled)
    return _hex_to_plain(restored)
