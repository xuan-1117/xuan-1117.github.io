def bytes_to_6bit_groups(data_bytes):
    """
    将字节串按6位一组进行分组（Base64编码的核心步骤）
    
    Args:
        data_bytes: 字节串（bytes）
    
    Returns:
        6位整数列表（0-63）
    """
    result = []
    buffer = 0
    bits_in_buffer = 0
    
    for byte in data_bytes:
        buffer = (buffer << 8) | byte
        bits_in_buffer += 8
        
        while bits_in_buffer >= 6:
            # 取高6位
            six_bits = (buffer >> (bits_in_buffer - 6)) & 0x3F
            result.append(six_bits)
            bits_in_buffer -= 6
            buffer &= (1 << bits_in_buffer) - 1  # 清空已取出的位
    
    # 处理剩余的位（可能需要填充）
    if bits_in_buffer > 0:
        # 左移补齐到6位
        six_bits = (buffer << (6 - bits_in_buffer)) & 0x3F
        result.append(six_bits)
    
    return result

def bytes_to_base64(data_bytes):
    """
    完整的Base64编码
    """
    # Base64字母表
    alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/"
    
    # 转换为6位组
    six_bit_groups = bytes_to_6bit_groups(data_bytes)
    
    # 转换为Base64字符
    base64_chars = [alphabet[group] for group in six_bit_groups]
    
    # 处理填充
    remainder = len(data_bytes) % 3
    if remainder == 1:
        base64_chars[-2:] = ['=', '=']
    elif remainder == 2:
        base64_chars[-1] = '='
    
    return ''.join(base64_chars)

# 示例
data = b"Hello, World!"
print(f"原始数据: {data}")
print(f"Base64编码: {bytes_to_base64(data)}")

def base64_to_bytes(base64_str):
    """
    Base64解码：将6位组转换回字节
    
    Args:
        base64_str: Base64编码的字符串
    
    Returns:
        解码后的字节串
    """
    # Base64字母表
    alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/"
    
    # 创建反向映射
    rev_alphabet = {char: idx for idx, char in enumerate(alphabet)}
    
    # 移除填充字符
    base64_str = base64_str.rstrip('=')
    
    # 将Base64字符转换为6位整数
    six_bit_groups = []
    for char in base64_str:
        if char in rev_alphabet:
            six_bit_groups.append(rev_alphabet[char])
        else:
            raise ValueError(f"无效的Base64字符: {char}")
    
    # 将6位组转换为字节
    result = bytearray()
    buffer = 0
    bits_in_buffer = 0
    
    for six_bits in six_bit_groups:
        buffer = (buffer << 6) | six_bits
        bits_in_buffer += 6
        
        while bits_in_buffer >= 8:
            # 取高8位
            byte = (buffer >> (bits_in_buffer - 8)) & 0xFF
            result.append(byte)
            bits_in_buffer -= 8
            buffer &= (1 << bits_in_buffer) - 1  # 清空已取出的位
    
    return bytes(result)

# 示例
base64_data = "SGVsbG8sIFdvcmxkIQ=="
print(f"Base64数据: {base64_data}")
print(f"解码结果: {base64_to_bytes(base64_data)}")

def binary_to_6bit_groups(binary_string):
    """
    将二进制字符串按6位一组进行分组
    
    Args:
        binary_string: 二进制字符串（如 '010101010101'）
    
    Returns:
        6位二进制字符串列表
    """
    # 确保长度是6的倍数，不足则补0
    padded_len = (len(binary_string) + 5) // 6 * 6
    padded_binary = binary_string.ljust(padded_len, '0')
    
    # 按6位分组
    groups = []
    for i in range(0, padded_len, 6):
        group = padded_binary[i:i+6]
        groups.append(group)
    
    return groups

def hex_to_6bit_groups(hex_string):
    """
    将十六进制字符串按6位一组进行分组
    """
    # 十六进制转二进制
    binary_string = bin(int(hex_string, 16))[2:]
    
    # 补齐到字节边界
    binary_len = len(hex_string) * 4
    binary_string = binary_string.zfill(binary_len)
    
    return binary_to_6bit_groups(binary_string)

# 示例
# 1. 二进制输入
binary_input = "010010010100000101001110"
print("二进制输入:", binary_input)
groups = binary_to_6bit_groups(binary_input)
print("6位分组:", groups)
print("对应的Base64字符:", [chr(65 + int(g, 2)) for g in groups])  # 简单映射

print("\n" + "="*60 + "\n")

# 2. 十六进制输入
hex_input = "48656C6C6F"  # "Hello"的十六进制
print("十六进制输入:", hex_input)
groups = hex_to_6bit_groups(hex_input)
print("6位分组:", groups)
print("分组值:", [int(g, 2) for g in groups])

def file_to_base64_groups(filename, chunk_size=24):
    """
    读取文件并显示其Base64编码的6位分组过程
    
    Args:
        filename: 文件名
        chunk_size: 每次处理的位数（建议为6的倍数，如24位=4个Base64字符）
    """
    try:
        with open(filename, 'rb') as f:
            data = f.read()
    except FileNotFoundError:
        print(f"文件不存在: {filename}")
        return
    
    print(f"文件大小: {len(data)} 字节")
    print(f"文件内容 (前100字节): {data[:100]}")
    print()
    
    # Base64字母表
    alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/"
    
    # 处理每个字节块
    for i in range(0, len(data), chunk_size // 8):
        chunk = data[i:i + chunk_size // 8]
        
        print(f"字节块 {i//3 + 1} (字节 {i} 到 {i + len(chunk) - 1}):")
        print(f"  Hex: {chunk.hex()}")
        
        # 转换为6位组
        groups = bytes_to_6bit_groups(chunk)
        
        print(f"  6位分组 ({len(groups)} 组):")
        for j, group in enumerate(groups):
            binary_str = bin(group)[2:].zfill(6)
            print(f"    分组 {j+1}: {binary_str} ({group:2d}) -> '{alphabet[group]}'")
        
        # 显示Base64编码
        base64_str = bytes_to_base64(chunk)
        print(f"  Base64编码: {base64_str}")
        print()

# 示例：处理文本文件
# 首先创建一个测试文件
with open("test.txt", "w") as f:
    f.write("Hello, Base64 encoding!")

file_to_base64_groups("test.txt")
