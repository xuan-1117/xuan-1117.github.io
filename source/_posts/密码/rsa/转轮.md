```plain
# 定义转轮，每个转轮是一个26个字母的字符串
wheels = [
    "ZWAXJGDLUBVIQHKYPNTCRMOSFE",  # 1
    "KPBELNACZDTRXMJQOYHGVSFUWI",  # 2
    "BDMAIZVRNSJUWFHTEQGYXPLOCK",  # 3
    "RPLNDVHGFCUKTEBSXQYIZMJWAO",  # 4
    "IHFRLABEUOTSGJVDKCPMNZQWXY",  # 5
    "AMKGHIWPNYCJBFZDRUSLOQXVET",  # 6
    "GWTHSPYBXIZULVKMRAFDCEONJQ",  # 7
    "NOZUTWDCVRJLXKISEFAPMYGHBQ",  # 8
    "XPLTDSRFHENYVUBMCQWAOIKZGJ",  # 9
    "UDNAJFBOWTGVRSCZQKELMXYIHP",  # 10
    "MNBVCXZQWERTPOIUYALSKDJFHG",  # 11
    "LVNCMXZPQOWEIURYTASBKJDFHG",  # 12
    "JZQAWSXCDERFVBGTYHNUMKILOP"   # 13
]

# 密钥顺序（转轮的使用顺序）
key_order = [2, 3, 7, 5, 13, 12, 9, 1, 8, 10, 4, 11, 6]  # 从1开始编号

# 密文
ciphertext = "NFQKSEVOQOFNP"

# 标准字母表（参考线）
alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"

# 解密过程
plaintext = ""
for j in range(len(ciphertext)):
    # 获取当前使用的转轮编号（转换为0-based索引）
    wheel_idx = key_order[j] - 1
    wheel = wheels[wheel_idx]
    
    # 当前密文字母
    cipher_char = ciphertext[j]
    
    # 找到密文字母在转轮中的位置
    r = wheel.index(cipher_char)
    
    # 遍历所有可能的偏移，寻找满足条件的明文字母
    found = False
    for i in range(26):
        # 计算旋转后的位置
        pos = (r + i) % 26
        # 如果旋转后该位置的字母等于标准字母表中对应位置的字母
        if wheel[pos] == alphabet[i]:
            plain_char = alphabet[i]
            plaintext += plain_char
            found = True
            break
    
    if not found:
        plaintext += "?"

print("解密结果:", plaintext)

# 另一种常见方法：将密文字母旋转到转轮开头，然后读取某一列
# 这里尝试读取所有转轮旋转后的第一列（索引0）
print("\n尝试另一种方法（旋转后取索引0的字母）：")
plaintext2 = ""
for j in range(len(ciphertext)):
    wheel_idx = key_order[j] - 1
    wheel = wheels[wheel_idx]
    cipher_char = ciphertext[j]
    
    # 旋转转轮，使密文字母出现在开头
    r = wheel.index(cipher_char)
    rotated_wheel = wheel[r:] + wheel[:r]
    
    # 取旋转后的第一个字母（索引0）
    # 但这样就是密文字母本身，所以没有意义
    # 因此我们尝试取固定偏移，比如偏移13
    offset = 13  # 尝试不同的偏移
    plain_char = rotated_wheel[offset]
    plaintext2 += plain_char

print(f"使用偏移{offset}的结果:", plaintext2)

# 尝试所有可能的偏移（0-25）
print("\n尝试所有可能的偏移（0-25）：")
for offset in range(26):
    plaintext_test = ""
    for j in range(len(ciphertext)):
        wheel_idx = key_order[j] - 1
        wheel = wheels[wheel_idx]
        cipher_char = ciphertext[j]
        
        r = wheel.index(cipher_char)
        rotated_wheel = wheel[r:] + wheel[:r]
        
        plain_char = rotated_wheel[offset]
        plaintext_test += plain_char
    
    print(f"偏移{offset:2d}: {plaintext_test}")
```
