import gmpy2
from Crypto.Util.number import long_to_bytes

# 给定的密文
c = 9217979941366220275377875095861710925207028551771520610387238734819759256223080175603032167658086669886661302962985046348865181740591251321966682848536331583243529

# 对c开平方
m_int, is_square = gmpy2.iroot(c, 2)

if is_square:
    m_int = int(m_int)
    print(f"解密后的整数: {m_int}")
    print(f"十六进制: {hex(m_int)}")
    
    # 转换为字节
    m_bytes = long_to_bytes(m_int)
    print(f"字节表示: {m_bytes}")
    
    # 尝试解码为字符串
    try:
        flag = m_bytes.decode('utf-8')
        print(f"Flag: {flag}")
    except UnicodeDecodeError:
        print("无法解码为UTF-8，可能是二进制数据或其他编码")
        print(f"原始字节的十六进制: {m_bytes.hex()}")
else:
    print("c不是完全平方数，无法直接开平方解密。")
